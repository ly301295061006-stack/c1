<script>
const avatar = document.getElementById('avatar');
const costumes = document.querySelectorAll('.costume');

costumes.forEach(c => {
  c.addEventListener('click', () => {
    avatar.style.transform = 'scale(1.2) rotate(10deg)'; // animation nhỏ
    setTimeout(()=>{ avatar.style.transform='scale(1) rotate(0deg)'; }, 300);
    avatar.src = c.src; // thay avatar mặc trang phục
  });
});

// Nút chia sẻ (demo)

});
